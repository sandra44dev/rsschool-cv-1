/*
Given two shuffled arrays with almost identical
values, except for one integer is missing in
one of them.
This function returns the missing integer.
*/
            
function findMissing(arr1, arr2) {
  let longArr = [];
  let shortArr = [];
            
  if (arr1.length > arr2.length){
    longArr = arr1;
    shortArr = arr2;
  } else {
  longArr = arr1;
  shortArr = arr2;
  }
            
  shortArr.map((number) => {
    longArr.splice(longArr.indexOf(number), 1);
  })
            
  return longArr[0];
}
